```python
import feedparser
import requests
from datetime import datetime
import json

class RSSAggregator:
    def __init__(self):
        self.feeds = {
            'openai': 'https://openai.com/blog/rss.xml',
            'anthropic': 'https://www.anthropic.com/news/rss',
            'google_ai': 'https://ai.googleblog.com/feeds/posts/default',
            'deepmind': 'https://deepmind.com/blog/feed/basic/',
            'arxiv': 'https://arxiv.org/rss/cs.AI',
            'huggingface': 'https://huggingface.co/blog/feed.xml'
        }
    
    def fetch_feed(self, url):
        """Fetch and parse RSS feed"""
        try:
            feed = feedparser.parse(url)
            return feed.entries
        except Exception as e:
            print(f"Error fetching {url}: {e}")
            return []
    
    def process_entries(self, entries, source):
        """Process RSS entries into standardized format"""
        processed = []
        for entry in entries:
            processed.append({
                'title': entry.get('title', ''),
                'link': entry.get('link', ''),
                'summary': entry.get('summary', ''),
                'published': entry.get('published', ''),
                'source': source,
                'timestamp': datetime.now().isoformat()
            })
        return processed
    
    def aggregate_news(self):
        """Aggregate all RSS feeds"""
        all_news = []
        for source, url in self.feeds.items():
            entries = self.fetch_feed(url)
            processed = self.process_entries(entries, source)
            all_news.extend(processed)
        
        # Sort by publication date
        all_news.sort(key=lambda x: x['published'], reverse=True)
        return all_news
    
    def save_news(self, news_data):
        """Save aggregated news to file"""
        with open('news_data.json', 'w') as f:
            json.dump(news_data, f, indent=2)

# Usage
aggregator = RSSAggregator()
news = aggregator.aggregate_news()
aggregator.save_news(news)
```

```javascript
const Parser = require('rss-parser');
const fs = require('fs');

class RSSAggregator {
    constructor() {
        this.parser = new Parser();
        this.feeds = {
            openai: 'https://openai.com/blog/rss.xml',
            anthropic: 'https://www.anthropic.com/news/rss',
            googleAI: 'https://ai.googleblog.com/feeds/posts/default',
            deepmind: 'https://deepmind.com/blog/feed/basic/',
            arxiv: 'https://arxiv.org/rss/cs.AI',
            huggingface: 'https://huggingface.co/blog/feed.xml'
        };
    }
    
    async fetchFeed(url) {
        try {
            const feed = await this.parser.parseURL(url);
            return feed.items;
        } catch (error) {
            console.error(`Error fetching ${url}:`, error);
            return [];
        }
    }
    
    processEntries(entries, source) {
        return entries.map(entry => ({
            title: entry.title || '',
            link: entry.link || '',
            summary: entry.contentSnippet || '',
            published: entry.pubDate || '',
            source: source,
            timestamp: new Date().toISOString()
        }));
    }
    
    async aggregateNews() {
        const allNews = [];
        
        for (const [source, url] of Object.entries(this.feeds)) {
            const entries = await this.fetchFeed(url);
            const processed = this.processEntries(entries, source);
            allNews.push(...processed);
        }
        
        // Sort by publication date
        allNews.sort((a, b) => new Date(b.published) - new Date(a.published));
        return allNews;
    }
    
    saveNews(newsData) {
        fs.writeFileSync('news_data.json', JSON.stringify(newsData, null, 2));
    }
}

// Usage
async function main() {
    const aggregator = new RSSAggregator();
    const news = await aggregator.aggregateNews();
    aggregator.saveNews(news);
}

main();
```
